Where to look for the Geminids

Select the location closest to you to see where in the night sky the Geminid radiant will be on the night of 14 December into the morning of 15 December.

import { aq, op } from "@uwdata/arquero"

regions = FileAttachment("regions.csv").csv()
regionTable = aq.from(regions)

regionMap = regionTable
  .derive({ map: d => [d.country + " - " + d.region, d.region] })
  .array("map")
viewof selectedRegion = Inputs.select(new Map(regionMap))

selectedRegionRow = regionTable
  .params({ selectedRegion: selectedRegion })
  .filter(d => d.region == selectedRegion)
  .objects()
video = html`<div style="padding:56.25% 0 0 0;position:relative;">
  <iframe
    src="https://player.vimeo.com/video/${selectedRegionRow[0].video_id_a}?h=${selectedRegionRow[0].video_id_b}&amp;badge=0&amp;autopause=0&amp;quality_selector=1&amp;player_id=0&amp;app_id=58479"
    frameborder="0"
    allow="autoplay; fullscreen; picture-in-picture"
    style="position:absolute; top:0; left:0; width:100%; height:100%; border-radius: 4px;"
    title="${selectedRegionRow.region}"></iframe>
</div>`
